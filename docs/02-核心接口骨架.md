## 一、统一返回模型 OperateResult

工业场景下，错误码、错误消息比抛异常更友好。所有可能失败的操作（通讯、IO、业务）都建议返回此类型。

### 1.1 OperateResult.cs

```csharp
namespace Acme.Industrial.Core.Results;

/// <summary>
/// 操作返回结果（无返回值）。
/// </summary>
public class OperateResult
{
    /// <summary>是否成功。</summary>
    public bool IsSuccess { get; set; }

    /// <summary>错误码（成功时为 0）。</summary>
    public int ErrorCode { get; set; }

    /// <summary>错误消息（成功时为 string.Empty）。</summary>
    public string Message { get; set; } = string.Empty;

    /// <summary>异常对象（仅在失败且来源是异常时填充）。</summary>
    public Exception? Exception { get; set; }

    /// <summary>耗时（毫秒）。</summary>
    public long ElapsedMilliseconds { get; set; }

    public OperateResult() { }

    public OperateResult(int errorCode, string message)
    {
        ErrorCode = errorCode;
        Message = message;
    }

    // ----------- 工厂方法 -----------

    public static OperateResult Ok() =>
        new() { IsSuccess = true };

    public static OperateResult Fail(int errorCode, string message) =>
        new() { IsSuccess = false, ErrorCode = errorCode, Message = message };

    public static OperateResult Fail(string message) =>
        Fail(-1, message);

    public static OperateResult Fail(Exception ex) =>
        new() { IsSuccess = false, ErrorCode = -1, Message = ex.Message, Exception = ex };

    // 类型转换
    public static OperateResult<T> Ok<T>(T content) =>
        new() { IsSuccess = true, Content = content };

    public static OperateResult<T> Fail<T>(int errorCode, string message) =>
        new() { IsSuccess = false, ErrorCode = errorCode, Message = message };

    public static OperateResult<T> Fail<T>(OperateResult source) =>
        new()
        {
            IsSuccess = false,
            ErrorCode = source.ErrorCode,
            Message = source.Message,
            Exception = source.Exception
        };

    public override string ToString() =>
        IsSuccess
            ? $"[OK] (Elapsed: {ElapsedMilliseconds}ms)"
            : $"[FAIL] Code={ErrorCode}, Msg={Message}";
}
```

### 1.3 使用示例

```csharp
public async Task<OperateResult<int>> ReadTemperatureAsync()
{
    try
    {
        var raw = await _modbus.ReadHoldingRegisterAsync(40001);
        if (!raw.IsSuccess)
            return OperateResult.Fail<int>(raw);

        return OperateResult.Ok(raw.Content);
    }
    catch (Exception ex)
    {
        return OperateResult.Fail<int>(-100, ex.Message);
    }
}

// 调用端
var result = await ReadTemperatureAsync();
if (!result.IsSuccess)
{
    _logger.Warn($"读温度失败: {result.ErrorCode} - {result.Message}");
    return;
}
Console.WriteLine($"温度 = {result.Content}");
```

### 2.4 模块加载策略（按需加载）

> 对于大型工业系统，模块众多，不可能一次性全部加载。采用**混合加载策略**可以平衡启动速度和资源占用。

```csharp
/// <summary>
/// 模块注册中心实现。
/// </summary>
public class ModuleRegistry : IModuleRegistry
{
    private readonly ConcurrentDictionary<string, ModuleMetadata> _modules = new();
    private readonly IAppLogger _logger;

    public ModuleRegistry(IAppLoggerFactory loggerFactory)
    {
        _logger = loggerFactory.CreateLogger<ModuleRegistry>();
    }

    public void Register(ModuleMetadata metadata)
    {
        if (_modules.TryAdd(metadata.Id, metadata))
        {
            _logger.Debug($"模块已注册: {metadata.Name} ({metadata.LoadPolicy})");
        }
        else
        {
            _logger.Warn($"模块重复注册: {metadata.Id}");
        }
    }

    public void RegisterRange(IEnumerable<ModuleMetadata> modules)
    {
        foreach (var metadata in modules)
            Register(metadata);
    }

    public IReadOnlyList<ModuleMetadata> GetAllMetadata() => _modules.Values.ToList();

    public ModuleMetadata? GetMetadata(string moduleId) =>
        _modules.TryGetValue(moduleId, out var m) ? m : null;

    public bool IsRegistered(string moduleId) => _modules.ContainsKey(moduleId);
}
```

```csharp
/// <summary>
/// 加载策略上下文。
/// </summary>
public class ModuleLoadContext
{
    /// <summary>模块注册中心。</summary>
    public required IModuleRegistry Registry { get; init; }
    
    /// <summary>服务解析器。</summary>
    public required IServiceResolver Services { get; init; }
    
    /// <summary>日志工厂。</summary>
    public IAppLoggerFactory LoggerFactory { get; init; } = null!;
    
    /// <summary>是否在启动阶段。</summary>
    public bool IsStartup { get; init; }
    
    /// <summary>已加载模块数量。</summary>
    public int LoadedCount { get; init; }
}

/// <summary>
/// 启动预加载策略 - 必须模块必须启动时加载。
/// </summary>
public class EagerLoadStrategy : IModuleLoadStrategy
{
    public bool ShouldLoadOnStartup(ModuleMetadata metadata)
        => metadata.LoadPolicy == ModuleLoadPolicy.Eager;
    
    public async Task<IModule?> LoadAsync(ModuleMetadata metadata, 
                                          ModuleLoadContext context, 
                                          CancellationToken ct = default)
    {
        return await LoadModuleInternalAsync(metadata, context, ct);
    }
}

/// <summary>
/// 懒加载策略 - 首次访问时加载。
/// </summary>
public class LazyLoadStrategy : IModuleLoadStrategy
{
    public bool ShouldLoadOnStartup(ModuleMetadata metadata) => false;
    
    public Task<IModule?> LoadAsync(ModuleMetadata metadata, 
                                    ModuleLoadContext context, 
                                    CancellationToken ct = default)
    {
        // 懒加载会在首次访问时触发
        return LoadModuleInternalAsync(metadata, context, ct);
    }
}

/// <summary>
/// 预加载策略 - 后台异步加载，不阻塞启动。
/// </summary>
public class PreloadStrategy : IModuleLoadStrategy
{
    public bool ShouldLoadOnStartup(ModuleMetadata metadata) => false;
    
    public Task<IModule?> LoadAsync(ModuleMetadata metadata, 
                                    ModuleLoadContext context, 
                                    CancellationToken ct = default)
    {
        // 后台加载，不阻塞
        return Task.Run(() => LoadModuleInternalAsync(metadata, context, ct), ct);
    }
}

/// <summary>
/// 手动加载策略 - 仅在显式调用时加载。
/// </summary>
public class ManualLoadStrategy : IModuleLoadStrategy
{
    public bool ShouldLoadOnStartup(ModuleMetadata metadata) => false;
    
    public Task<IModule?> LoadAsync(ModuleMetadata metadata, 
                                    ModuleLoadContext context, 
                                    CancellationToken ct = default)
    {
        return LoadModuleInternalAsync(metadata, context, ct);
    }
}

internal static class ModuleLoadStrategyExtensions
{
    internal static async Task<IModule?> LoadModuleInternalAsync(
        ModuleMetadata metadata, 
        ModuleLoadContext context, 
        CancellationToken ct)
    {
        var logger = context.LoggerFactory.CreateLogger($"Module.{metadata.Id}");
        
        try
        {
            // 1. 检查依赖
            foreach (var depId in metadata.Dependencies)
            {
                var depMeta = context.Registry.GetMetadata(depId);
                if (depMeta == null)
                {
                    logger.Error($"模块 {metadata.Id} 依赖的模块不存在: {depId}");
                    return null;
                }
            }
            
            // 2. 加载程序集
            var assembly = LoadAssembly(metadata.AssemblyPath);
            if (assembly == null)
            {
                logger.Error($"加载程序集失败: {metadata.AssemblyPath}");
                return null;
            }
            
            // 3. 创建实例
            var moduleType = assembly.GetType(metadata.EntryType);
            if (moduleType == null)
            {
                logger.Error($"找不到模块类型: {metadata.EntryType}");
                return null;
            }
            
            var module = (IModule?)Activator.CreateInstance(moduleType);
            if (module == null)
            {
                logger.Error($"创建模块实例失败: {metadata.Id}");
                return null;
            }
            
            // 4. 注册服务
            module.RegisterServices(context.Services);
            
            // 5. 初始化
            await module.OnInitializeAsync(context.Services, ct);
            
            logger.Info($"模块已加载: {metadata.Name} v{metadata.Version}");
            return module;
        }
        catch (Exception ex)
        {
            logger.Error($"模块加载失败: {metadata.Name}", ex);
            return null;
        }
    }
    
    private static Assembly? LoadAssembly(string path)
    {
        var probingPaths = new[]
        {
            path,
            Path.Combine(AppDomain.CurrentDomain.BaseDirectory, "Modules", Path.GetFileName(path)),
            Path.Combine(AppDomain.CurrentDomain.BaseDirectory, Path.GetFileName(path))
        };

        foreach (var probingPath in probingPaths)
        {
            if (File.Exists(probingPath))
                return Assembly.LoadFrom(probingPath);
        }
        return null;
    }
}
```

```csharp
/// <summary>
/// 模块管理器实现。
/// </summary>
public class ModuleManager : IModuleManager
{
    private readonly IModuleRegistry _registry;
    private readonly IServiceResolver _services;
    private readonly IAppLogger _logger;
    private readonly ConcurrentDictionary<string, ModuleInstance> _instances = new();
    private readonly ConcurrentDictionary<string, Task<IModule?>> _loadingTasks = new();

    public event EventHandler<ModuleStateChangedEventArgs>? StateChanged;

    public ModuleManager(
        IModuleRegistry registry,
        IServiceResolver services,
        IAppLoggerFactory loggerFactory)
    {
        _registry = registry;
        _services = services;
        _logger = loggerFactory.CreateLogger<ModuleManager>();
    }

    public IModule? GetModule(string moduleId) =>
        _instances.TryGetValue(moduleId, out var inst) ? inst.Module : null;

    public IReadOnlyList<IModule> GetLoadedModules() =>
        _instances.Values.Select(i => i.Module).ToList();

    public bool IsLoaded(string moduleId) => _instances.ContainsKey(moduleId);

    public async Task<OperateResult<IModule>> LoadModuleAsync(
        string moduleId, CancellationToken ct = default)
    {
        // 已加载则直接返回
        if (_instances.TryGetValue(moduleId, out var existing))
            return OperateResult.Ok(existing.Module);

        // 正在加载则等待
        if (_loadingTasks.TryGetValue(moduleId, out var loadingTask))
        {
            var module = await loadingTask;
            return module != null 
                ? OperateResult.Ok(module) 
                : OperateResult.Fail<IModule>($"模块加载失败: {moduleId}");
        }

        var metadata = _registry.GetMetadata(moduleId);
        if (metadata == null)
            return OperateResult.Fail<IModule>($"模块未注册: {moduleId}");

        var context = CreateContext();
        var loadTask = DoLoadAsync(metadata, context, ct);
        _loadingTasks[moduleId] = loadTask;

        try
        {
            var module = await loadTask;
            if (module != null)
                return OperateResult.Ok(module);
            return OperateResult.Fail<IModule>($"模块加载失败: {moduleId}");
        }
        finally
        {
            _loadingTasks.TryRemove(moduleId, out _);
        }
    }

    private async Task<IModule?> DoLoadAsync(
        ModuleMetadata metadata, 
        ModuleLoadContext context, 
        CancellationToken ct)
    {
        RaiseStateChanged(metadata, ModuleState.Loading);

        try
        {
            // 递归加载依赖
            foreach (var depId in metadata.Dependencies)
            {
                if (!_instances.ContainsKey(depId))
                {
                    var depMeta = _registry.GetMetadata(depId);
                    if (depMeta != null)
                    {
                        await DoLoadAsync(depMeta, context, ct);
                    }
                }
            }

            RaiseStateChanged(metadata, ModuleState.Initializing);

            var assembly = LoadAssembly(metadata.AssemblyPath);
            if (assembly == null) return null;

            var moduleType = assembly.GetType(metadata.EntryType);
            if (moduleType == null) return null;

            var module = (IModule?)Activator.CreateInstance(moduleType);
            if (module == null) return null;

            module.RegisterServices(_services);
            await module.OnInitializeAsync(_services, ct);

            var instance = new ModuleInstance(metadata, module);
            _instances[metadata.Id] = instance;

            RaiseStateChanged(metadata, ModuleState.Loaded, "模块加载成功");
            _logger.Info($"模块已加载: {metadata.Name}");

            return module;
        }
        catch (Exception ex)
        {
            _logger.Error($"模块加载失败: {metadata.Name}", ex);
            RaiseStateChanged(metadata, ModuleState.Unloaded, ex.Message);
            return null;
        }
    }

    public async Task<OperateResult> UnloadModuleAsync(
        string moduleId, CancellationToken ct = default)
    {
        if (!_instances.TryGetValue(moduleId, out var instance))
            return OperateResult.Fail($"模块未加载: {moduleId}");

        try
        {
            RaiseStateChanged(instance.Metadata, ModuleState.Deactivating);
            
            await instance.Module.OnShutdownAsync(ct);
            instance.Dispose();
            _instances.TryRemove(moduleId, out _);

            RaiseStateChanged(instance.Metadata, ModuleState.Unloaded, "模块已卸载");
            return OperateResult.Ok();
        }
        catch (Exception ex)
        {
            _logger.Error($"模块卸载失败: {moduleId}", ex);
            return OperateResult.Fail(ex.Message);
        }
    }

    public async Task PreloadAllAsync(CancellationToken ct = default)
    {
        var preloadModules = _registry.GetAllMetadata()
            .Where(m => m.LoadPolicy == ModuleLoadPolicy.Preload);

        foreach (var metadata in preloadModules)
        {
            if (ct.IsCancellationRequested) break;
            if (!_instances.ContainsKey(metadata.Id))
            {
                await LoadModuleAsync(metadata.Id, ct);
            }
        }
    }

    private ModuleLoadContext CreateContext() => new()
    {
        Registry = _registry,
        Services = _services,
        LoggerFactory = _services.Resolve<IAppLoggerFactory>()!,
        IsStartup = false,
        LoadedCount = _instances.Count
    };

    private Assembly? LoadAssembly(string path)
    {
        var probingPaths = new[]
        {
            path,
            Path.Combine(AppDomain.CurrentDomain.BaseDirectory, "Modules", Path.GetFileName(path))
        };

        foreach (var probingPath in probingPaths)
        {
            if (File.Exists(probingPath))
                return Assembly.LoadFrom(probingPath);
        }
        return null;
    }

    private void RaiseStateChanged(ModuleMetadata metadata, ModuleState state, string? message = null)
    {
        StateChanged?.Invoke(this, new ModuleStateChangedEventArgs
        {
            ModuleId = metadata.Id,
            ModuleName = metadata.Name,
            OldState = ModuleState.Registered,
            NewState = state,
            Message = message
        });
    }

    public async ValueTask DisposeAsync()
    {
        foreach (var instance in _instances.Values.ToList())
        {
            await instance.Module.OnShutdownAsync(CancellationToken.None);
            instance.Dispose();
        }
        _instances.Clear();
    }
}
```

#### 2.4.6 启动引导器

```csharp
namespace Acme.Industrial.Core.Bootstrap;

/// <summary>
/// 应用程序启动器。
/// </summary>
public class ApplicationBootstrap
{
    private readonly IModuleRegistry _registry;
    private readonly IModuleManager _manager;
    private readonly IModuleLoadStrategy _eagerStrategy;
    private readonly IAppLogger _logger;

    public ApplicationBootstrap(
        IModuleRegistry registry,
        IModuleManager manager,
        IAppLoggerFactory loggerFactory)
    {
        _registry = registry;
        _manager = manager;
        _eagerStrategy = new EagerLoadStrategy();
        _logger = loggerFactory.CreateLogger<ApplicationBootstrap>();
    }

    /// <summary>
    /// 启动应用程序。
    /// </summary>
    public async Task StartAsync(CancellationToken ct = default)
    {
        _logger.Info("=== 应用启动 ===");

        // 1. 扫描模块
        var modules = ScanModules();
        _registry.RegisterRange(modules);
        _logger.Info($"发现 {modules.Count} 个模块");

        // 2. 拓扑排序（按依赖关系排序）
        var sorted = TopologicalSort(modules);

        // 3. 加载启动必须模块（Eager）
        var context = new ModuleLoadContext
        {
            Registry = _registry,
            Services = null!,  // 启动时服务尚未构建
            IsStartup = true,
            LoadedCount = 0
        };

        foreach (var metadata in sorted.Where(m => m.LoadPolicy == ModuleLoadPolicy.Eager))
        {
            _logger.Debug($"加载模块: {metadata.Name}");
            await _manager.LoadModuleAsync(metadata.Id, ct);
        }

        // 4. 后台预加载（Preload）
        _ = _manager.PreloadAllAsync(ct);

        _logger.Info("=== 应用启动完成 ===");
    }

    private List<ModuleMetadata> ScanModules()
    {
        var modules = new List<ModuleMetadata>();
        var searchPaths = new[]
        {
            Path.Combine(AppDomain.CurrentDomain.BaseDirectory, "Modules"),
            Path.Combine(AppDomain.CurrentDomain.BaseDirectory, "Plugins")
        };

        foreach (var searchPath in searchPaths)
        {
            if (!Directory.Exists(searchPath)) continue;

            foreach (var file in Directory.GetFiles(searchPath, "*.dll"))
            {
                try
                {
                    var metadata = ExtractMetadata(file);
                    if (metadata != null)
                        modules.Add(metadata);
                }
                catch (Exception ex)
                {
                    _logger.Warn($"扫描模块失败: {file}", ex);
                }
            }
        }

        return modules;
    }

    private ModuleMetadata? ExtractMetadata(string dllPath)
    {
        Assembly? assembly;
        try
        {
            assembly = Assembly.LoadFrom(dllPath);
        }
        catch
        {
            return null;
        }

        var entryType = assembly.GetTypes()
            .FirstOrDefault(t => typeof(IModule).IsAssignableFrom(t) && !t.IsInterface);

        if (entryType == null) return null;

        var module = (IModule?)Activator.CreateInstance(entryType);
        if (module == null) return null;

        // 从特性读取加载策略
        var loadPolicy = ModuleLoadPolicy.Lazy;
        var attr = entryType.GetCustomAttribute<ModuleLoadPolicyAttribute>();
        if (attr != null)
            loadPolicy = attr.Policy;

        return new ModuleMetadata
        {
            Id = module.Id,
            Name = module.Name,
            Version = module.Version,
            EntryType = entryType.FullName ?? "",
            AssemblyPath = dllPath,
            Dependencies = module.Dependencies,
            LoadPolicy = loadPolicy
        };
    }

    private List<ModuleMetadata> TopologicalSort(List<ModuleMetadata> modules)
    {
        var result = new List<ModuleMetadata>();
        var visited = new HashSet<string>();
        var moduleDict = modules.ToDictionary(m => m.Id);

        void Visit(ModuleMetadata m)
        {
            if (visited.Contains(m.Id)) return;
            visited.Add(m.Id);

            foreach (var depId in m.Dependencies)
            {
                if (moduleDict.TryGetValue(depId, out var dep))
                    Visit(dep);
            }

            result.Add(m);
        }

        foreach (var m in modules)
            Visit(m);

        return result;
    }
}

/// <summary>
/// 模块加载策略特性。
/// </summary>
[AttributeUsage(AttributeTargets.Class)]
public class ModuleLoadPolicyAttribute : Attribute
{
    public ModuleLoadPolicy Policy { get; }
    public ModuleLoadPolicyAttribute(ModuleLoadPolicy policy) => Policy = policy;
}
```

#### 2.4.9 关键设计模式总结

| 设计模式 | 用途 |
|----------|------|
| **注册表模式 (Registry)** | 模块元数据统一管理，启动时扫描 |
| **策略模式 (Strategy)** | 不同加载策略（Eager/Lazy/Preload）可替换 |
| **单例模式 (Singleton)** | ModuleManager 全局唯一 |
| **工厂模式 (Factory)** | 模块实例创建 |
| **观察者模式 (Observer)** | 模块状态变更通知 |
| **拓扑排序** | 依赖顺序加载 |
| **异步加载** | Preload 策略不阻塞启动 |
| **依赖注入** | 服务解析 |

### 15.1 Autofac 包装器

```csharp
namespace Acme.Industrial.Infrastructure.DI;

/// <summary>
/// Autofac DI 容器包装器。
/// </summary>
public class AutofacServiceContainer : IServiceRegistry, IServiceResolver
{
    private readonly ContainerBuilder _builder = new();
    private readonly List<Type> _registeredTypes = new();
    private IContainer? _container;
    private bool _isLocked;

    public void Register<TService, TImpl>(ServiceLifetime lifetime = ServiceLifetime.Singleton)
        where TImpl : class, TService
    {
        CheckNotLocked();
        var rb = GetRegistrationBuilder<TService, TImpl>();
        switch (lifetime)
        {
            case ServiceLifetime.Singleton:
                rb.SingleInstance();
                break;
            case ServiceLifetime.Scoped:
                rb.InstancePerLifetimeScope();
                break;
            case ServiceLifetime.Transient:
                rb.InstancePerDependency();
                break;
        }
        _registeredTypes.Add(typeof(TService));
    }

    public void Register<TService>(TService instance) where TService : class
    {
        CheckNotLocked();
        _builder.RegisterInstance(instance).As<TService>();
        _registeredTypes.Add(typeof(TService));
    }

    public void Register(Type service, Type implementation, ServiceLifetime lifetime)
    {
        CheckNotLocked();
        var rb = _builder.RegisterType(implementation).As(service);
        switch (lifetime)
        {
            case ServiceLifetime.Singleton:
                rb.SingleInstance();
                break;
            case ServiceLifetime.Scoped:
                rb.InstancePerLifetimeScope();
                break;
            case ServiceLifetime.Transient:
                rb.InstancePerDependency();
                break;
        }
        _registeredTypes.Add(service);
    }

    public T Resolve<T>()
    {
        EnsureContainer();
        return _container!.Resolve<T>();
    }

    public T? TryResolve<T>()
    {
        EnsureContainer();
        return _container!.TryResolve<T>(out var result) ? result : default;
    }

    public IEnumerable<T> ResolveAll<T>()
    {
        EnsureContainer();
        return _container!.Resolve<IEnumerable<T>>();
    }

    public IServiceResolver CreateScope()
    {
        EnsureContainer();
        return new ScopeResolver(_container!.BeginLifetimeScope());
    }

    /// <summary>
    /// 从模块注册服务。
    /// </summary>
    public void RegisterFromModule(IModule module)
    {
        module.RegisterServices(this);
    }

    /// <summary>
    /// 构建容器。
    /// </summary>
    public void Build()
    {
        if (_container != null) return;
        _container = _builder.Build();
        _isLocked = true;
    }

    private void CheckNotLocked()
    {
        if (_isLocked)
            throw new InvalidOperationException("容器已构建，禁止注册");
    }

    private void EnsureContainer()
    {
        if (_container == null)
            Build();
    }

    private RegistrationBuilder<TService, TImpl, ScarlettContent> GetRegistrationBuilder<TService, TImpl>()
        where TImpl : class, TService
    {
        return _builder.RegisterType<TImpl>().As<TService>();
    }

    public void Dispose()
    {
        _container?.Dispose();
    }

    private class ScopeResolver : IServiceResolver
    {
        private readonly ILifetimeScope _scope;

        public ScopeResolver(ILifetimeScope scope)
        {
            _scope = scope;
        }

        public T Resolve<T>() => _scope.Resolve<T>();

        public T? TryResolve<T>() => _scope.TryResolve<T>(out var result) ? result : default;

        public IEnumerable<T> ResolveAll<T>() => _scope.Resolve<IEnumerable<T>>();

        public IServiceResolver CreateScope() => new ScopeResolver(_scope.BeginLifetimeScope());

        public void Dispose() => _scope.Dispose();
    }
}
```

### 16.1 Repository 接口定义

```csharp
namespace Acme.Industrial.Core.Data;

/// <summary>
/// 通用仓储接口。
/// </summary>
public interface IRepository<TEntity, TKey> where TEntity : class
{
    // ========== 查询 ==========
    Task<TEntity?> GetByIdAsync(TKey id, CancellationToken ct = default);
    Task<IReadOnlyList<TEntity>> GetAllAsync(CancellationToken ct = default);
    Task<TEntity?> FirstOrDefaultAsync(Expression<Func<TEntity, bool>> predicate, 
                                       CancellationToken ct = default);
    Task<IReadOnlyList<TEntity>> FindAsync(Expression<Func<TEntity, bool>> predicate,
                                            CancellationToken ct = default);
    
    // ========== 分页 ==========
    Task<PagedResult<TEntity>> GetPageAsync(int pageIndex, int pageSize,
                                             Expression<Func<TEntity, bool>>? predicate = null,
                                             Func<IQueryable<TEntity>, IOrderedQueryable<TEntity>>? orderBy = null,
                                             CancellationToken ct = default);
    
    // ========== 聚合 ==========
    Task<int> CountAsync(Expression<Func<TEntity, bool>>? predicate = null,
                         CancellationToken ct = default);
    Task<bool> ExistsAsync(Expression<Func<TEntity, bool>> predicate,
                           CancellationToken ct = default);
    
    // ========== 新增 ==========
    Task<TEntity> AddAsync(TEntity entity, CancellationToken ct = default);
    Task<IReadOnlyList<TEntity>> AddRangeAsync(IEnumerable<TEntity> entities,
                                                CancellationToken ct = default);
    
    // ========== 更新 ==========
    Task UpdateAsync(TEntity entity, CancellationToken ct = default);
    Task UpdateRangeAsync(IEnumerable<TEntity> entities, CancellationToken ct = default);
    
    // ========== 删除 ==========
    Task DeleteAsync(TKey id, CancellationToken ct = default);
    Task DeleteAsync(TEntity entity, CancellationToken ct = default);
    Task DeleteRangeAsync(IEnumerable<TEntity> entities, CancellationToken ct = default);
    
    // ========== 执行 ==========
    Task<int> ExecuteAsync(string sql, CancellationToken ct = default);
    Task<IReadOnlyList<T>> ExecuteQueryAsync<T>(string sql, CancellationToken ct = default) where T : class;
    
    // ========== 事务 ==========
    Task<OperateResult> ExecuteInTransactionAsync(Func<Task> action, 
                                                  CancellationToken ct = default);
}

/// <summary>
/// 泛型仓储实现基类（基于 SqlSugar）。
/// </summary>
public abstract class RepositoryBase<TEntity, TKey> : IRepository<TEntity, TKey> 
    where TEntity : class
{
    protected readonly ISqlSugarClient _db;
    protected readonly IAppLogger _logger;

    protected RepositoryBase(ISqlSugarClient db, IAppLoggerFactory loggerFactory)
    {
        _db = db;
        _logger = loggerFactory.CreateLogger<IRepository<TEntity, TKey>>();
    }

    public virtual async Task<TEntity?> GetByIdAsync(TKey id, CancellationToken ct = default)
    {
        return await _db.Queryable<TEntity>()
            .Where(new { Id = id })
            .FirstAsync(ct);
    }

    public virtual async Task<IReadOnlyList<TEntity>> GetAllAsync(CancellationToken ct = default)
    {
        return await _db.Queryable<TEntity>().ToListAsync(ct);
    }

    public virtual async Task<TEntity?> FirstOrDefaultAsync(
        Expression<Func<TEntity, bool>> predicate, CancellationToken ct = default)
    {
        return await _db.Queryable<TEntity>()
            .Where(predicate)
            .FirstAsync(ct);
    }

    public virtual async Task<IReadOnlyList<TEntity>> FindAsync(
        Expression<Func<TEntity, bool>> predicate, CancellationToken ct = default)
    {
        return await _db.Queryable<TEntity>()
            .Where(predicate)
            .ToListAsync(ct);
    }

    public virtual async Task<PagedResult<TEntity>> GetPageAsync(
        int pageIndex, int pageSize,
        Expression<Func<TEntity, bool>>? predicate = null,
        Func<IQueryable<TEntity>, IOrderedQueryable<TEntity>>? orderBy = null,
        CancellationToken ct = default)
    {
        var query = _db.Queryable<TEntity>();
        
        if (predicate != null)
            query = query.Where(predicate);
        
        if (orderBy != null)
            query = (ISugarQueryable<TEntity>)orderBy(query);

        var total = await query.CountAsync(ct);
        var items = await query
            .ToPageListAsync(pageIndex + 1, pageSize, total, ct);

        return new PagedResult<TEntity>
        {
            Items = items,
            TotalCount = total,
            PageIndex = pageIndex,
            PageSize = pageSize
        };
    }

    public virtual async Task<int> CountAsync(
        Expression<Func<TEntity, bool>>? predicate = null, 
        CancellationToken ct = default)
    {
        var query = _db.Queryable<TEntity>();
        if (predicate != null)
            query = query.Where(predicate);
        return await query.CountAsync(ct);
    }

    public virtual async Task<bool> ExistsAsync(
        Expression<Func<TEntity, bool>> predicate, CancellationToken ct = default)
    {
        return await _db.Queryable<TEntity>().Where(predicate).AnyAsync(ct);
    }

    public virtual async Task<TEntity> AddAsync(TEntity entity, CancellationToken ct = default)
    {
        var id = await _db.Insertable(entity).ExecuteReturnEntityAsync(ct);
        _logger.Debug($"新增实体: {typeof(TEntity).Name}");
        return id;
    }

    public virtual async Task<IReadOnlyList<TEntity>> AddRangeAsync(
        IEnumerable<TEntity> entities, CancellationToken ct = default)
    {
        var list = entities.ToList();
        await _db.Insertable(list).ExecuteCommandAsync(ct);
        _logger.Debug($"批量新增: {list.Count} 条");
        return list;
    }

    public virtual async Task UpdateAsync(TEntity entity, CancellationToken ct = default)
    {
        await _db.Updateable(entity).ExecuteCommandAsync(ct);
        _logger.Debug($"更新实体: {typeof(TEntity).Name}");
    }

    public virtual async Task UpdateRangeAsync(
        IEnumerable<TEntity> entities, CancellationToken ct = default)
    {
        await _db.Updateable(entities.ToList()).ExecuteCommandAsync(ct);
    }

    public virtual async Task DeleteAsync(TKey id, CancellationToken ct = default)
    {
        await _db.Deleteable<TEntity>().Where(new { Id = id }).ExecuteCommandAsync(ct);
        _logger.Debug($"删除实体: {typeof(TEntity).Name}, Id={id}");
    }

    public virtual async Task DeleteAsync(TEntity entity, CancellationToken ct = default)
    {
        await _db.Deleteable(entity).ExecuteCommandAsync(ct);
    }

    public virtual async Task DeleteRangeAsync(
        IEnumerable<TEntity> entities, CancellationToken ct = default)
    {
        await _db.Deleteable(entities.ToList()).ExecuteCommandAsync(ct);
    }

    public virtual async Task<int> ExecuteAsync(string sql, CancellationToken ct = default)
    {
        return await _db.Ado.ExecuteCommandAsync(sql, ct: ct);
    }

    public virtual async Task<IReadOnlyList<T>> ExecuteQueryAsync<T>(
        string sql, CancellationToken ct = default) where T : class
    {
        return await _db.SqlQueryable<T>(sql).ToListAsync(ct);
    }

    public virtual async Task<OperateResult> ExecuteInTransactionAsync(
        Func<Task> action, CancellationToken ct = default)
    {
        try
        {
            await _db.Ado.UseTranAsync(async () => await action());
            return OperateResult.Ok();
        }
        catch (Exception ex)
        {
            _logger.Error("事务执行失败", ex);
            return OperateResult.Fail(ex.Message);
        }
    }
}
```

### 16.2 单元OfWork模式

```csharp
namespace Acme.Industrial.Core.Data;

/// <summary>
/// 工作单元接口（事务管理）。
/// </summary>
public interface IUnitOfWork : IDisposable
{
    /// <summary>获取指定类型的仓储。</summary>
    IRepository<TEntity, TKey> GetRepository<TEntity, TKey>() where TEntity : class;
    
    /// <summary>开始事务。</summary>
    Task BeginAsync(CancellationToken ct = default);
    
    /// <summary>提交事务。</summary>
    Task CommitAsync(CancellationToken ct = default);
    
    /// <summary>回滚事务。</summary>
    Task RollbackAsync(CancellationToken ct = default);
    
    /// <summary>保存更改。</summary>
    Task<int> SaveChangesAsync(CancellationToken ct = default);
}

/// <summary>
/// 工作单元实现。
/// </summary>
public class UnitOfWork : IUnitOfWork
{
    private readonly ISqlSugarClient _db;
    private readonly ConcurrentDictionary<Type, object> _repositories = new();
    private readonly IAppLogger _logger;
    private bool _disposed;

    public UnitOfWork(ISqlSugarClient db, IAppLoggerFactory loggerFactory)
    {
        _db = db;
        _logger = loggerFactory.CreateLogger<UnitOfWork>();
    }

    public IRepository<TEntity, TKey> GetRepository<TEntity, TKey>() 
        where TEntity : class
    {
        var type = typeof(TEntity);
        if (!_repositories.TryGetValue(type, out var repo))
        {
            repo = CreateRepository<TEntity, TKey>();
            _repositories[type] = repo;
        }
        return (IRepository<TEntity, TKey>)repo;
    }

    protected virtual IRepository<TEntity, TKey> CreateRepository<TEntity, TKey>()
        where TEntity : class
    {
        var repoType = typeof(RepositoryBase<TEntity, TKey>);
        return (IRepository<TEntity, TKey>)Activator.CreateInstance(repoType, _db, null!)!;
    }

    public Task BeginAsync(CancellationToken ct = default)
    {
        return Task.CompletedTask; // SqlSugar 自动管理事务
    }

    public async Task CommitAsync(CancellationToken ct = default)
    {
        await _db.Ado.CommitTranAsync();
        _logger.Debug("事务已提交");
    }

    public async Task RollbackAsync(CancellationToken ct = default)
    {
        await _db.Ado.RollbackTranAsync();
        _logger.Debug("事务已回滚");
    }

    public Task<int> SaveChangesAsync(CancellationToken ct = default)
    {
        return Task.FromResult(0); // SqlSugar 自动保存
    }

    public void Dispose()
    {
        if (_disposed) return;
        _disposed = true;
    }
}
```

## 十八、设计原则总结

| 原则 | 说明 |
|------|------|
| **接口先行** | 每个能力先定义接口，再做实现，方便替换与测试 |
| **OperateResult 优先** | 可预期的失败用返回值，不可预期的才抛异常 |
| **异步到底** | 所有 IO/通讯/数据库操作均提供 Async 版本，配合 CancellationToken |
| **依赖抽象** | 业务代码不依赖 Serilog/Autofac/SqlSugar 等具体实现 |
| **不可变模型** | DTO/Event 使用 init 属性，避免被意外修改 |
| **职责单一** | 一个接口只做一件事；接口臃肿时拆分 |
| **可扩展点显式** | 通过事件总线、模块/插件机制、脚本引擎、HMI扩展点提供扩展点 |
| **Repository 模式** | 数据访问通过仓储抽象，支持多种数据库 |
| **UnitOfWork** | 事务管理通过工作单元，确保数据一致性 |
| **模块化 DI** | Autofac 容器支持模块化注册和生命周期管理 |
