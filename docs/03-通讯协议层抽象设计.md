# 工业级 WinForm 框架 —— 通讯协议层抽象设计

> 本文档是工业框架最关键的部分。它定义了**统一的设备驱动接口**、**点位（Tag）模型**、**连接管理**、**字节序处理**、**OperateResult 错误模型**等核心抽象，让上层业务无需关心底层是 Modbus 还是 S7。

---

## 一、设计目标

| 目标 | 说明 |
|------|------|
| **协议无关** | 业务层只面向 `IDeviceDriver`，不直接依赖 Modbus/S7/MC 等具体实现 |
| **异步优先** | 全异步 API，避免 UI 阻塞 |
| **统一错误处理** | 所有读写返回 `OperateResult<T>`，不抛异常 |
| **批量优化** | 支持批量读写、自动合并相邻地址 |
| **订阅式刷新** | 支持点位订阅 + 周期采集 + 变化通知 |
| **脚本集成** | 上层脚本引擎和 HMI 系统可通过统一接口访问点位 |
| **可测试** | 提供 Mock 驱动，便于无设备开发 |
| **可扩展** | 新增协议只需实现接口，宿主无需改动 |
| **可观测** | 内置统计：读写次数、成功率、延时、最后错误 |

## 四、抽象基类（减少各协议重复代码）

```csharp
namespace Acme.Industrial.Communication.Core;

/// <summary>
/// 所有协议驱动的抽象基类，封装连接状态管理、重试、统计、心跳、断线重连。
/// </summary>
public abstract class DeviceDriverBase : IDeviceDriver
{
    private readonly SemaphoreSlim _connLock = new(1, 1);
    private CancellationTokenSource? _heartbeatCts;
    protected IAppLogger Logger { get; }
    protected IByteTransform ByteTransform { get; }

    public string DeviceId => Options.DeviceId;
    public string Protocol => Options.Protocol;
    public ConnectionOptions Options { get; }
    public ConnectionState State { get; private set; } = ConnectionState.Disconnected;
    public CommunicationStatistics Statistics { get; } = new();

    public event EventHandler<ConnectionStateChangedEventArgs>? StateChanged;
    public event EventHandler<CommunicationErrorEventArgs>? ErrorOccurred;

    protected DeviceDriverBase(ConnectionOptions options, IAppLoggerFactory loggerFactory,
                               IByteTransform byteTransform)
    {
        Options = options;
        Logger = loggerFactory.CreateLogger($"Driver.{options.Protocol}.{options.DeviceId}");
        ByteTransform = byteTransform;
    }

    // ----------- 模板方法（子类实现） -----------
    protected abstract Task<OperateResult> ConnectCoreAsync(CancellationToken ct);
    protected abstract Task<OperateResult> DisconnectCoreAsync(CancellationToken ct);
    protected abstract Task<OperateResult<byte[]>> ReadRawCoreAsync(
        string address, ushort length, CancellationToken ct);
    protected abstract Task<OperateResult> WriteRawCoreAsync(
        string address, byte[] data, CancellationToken ct);
    protected abstract Task<OperateResult> PingCoreAsync(CancellationToken ct);

    // ----------- 公共实现 -----------
    public async Task<OperateResult> ConnectAsync(CancellationToken ct = default)
    {
        await _connLock.WaitAsync(ct);
        try
        {
            if (State == ConnectionState.Connected) return OperateResult.Ok();

            ChangeState(ConnectionState.Connecting);
            var result = await WithRetry(() => ConnectCoreAsync(ct), Options.RetryCount, ct);

            if (result.IsSuccess)
            {
                ChangeState(ConnectionState.Connected);
                Statistics.ConnectedSince = DateTime.Now;
                StartHeartbeat();
            }
            else
            {
                ChangeState(ConnectionState.Faulted);
            }
            return result;
        }
        finally { _connLock.Release(); }
    }

    public async Task<OperateResult> DisconnectAsync(CancellationToken ct = default)
    {
        StopHeartbeat();
        await _connLock.WaitAsync(ct);
        try
        {
            var result = await DisconnectCoreAsync(ct);
            ChangeState(ConnectionState.Disconnected);
            return result;
        }
        finally { _connLock.Release(); }
    }

    public Task<OperateResult> PingAsync(CancellationToken ct = default) =>
        WithRetry(() => PingCoreAsync(ct), 1, ct);

    public async Task<OperateResult<byte[]>> ReadRawAsync(
        string address, ushort length, CancellationToken ct = default)
    {
        if (State != ConnectionState.Connected)
            return OperateResult.Fail<byte[]>(
                ErrorCode.CommNotConnected, "Device not connected.");

        var sw = Stopwatch.StartNew();
        var result = await WithRetry(() => ReadRawCoreAsync(address, length, ct),
                                     Options.RetryCount, ct);
        sw.Stop();

        Statistics.TotalReads++;
        if (result.IsSuccess)
        {
            Statistics.SuccessReads++;
            Statistics.TotalBytesReceived += result.Content?.Length ?? 0;
            Statistics.LastSuccessTime = DateTime.Now;
            UpdateAvgLatency(sw.ElapsedMilliseconds);
        }
        else
        {
            Statistics.FailedReads++;
            Statistics.LastErrorTime = DateTime.Now;
            Statistics.LastErrorMessage = result.Message;
            OnError(result);
        }
        result.ElapsedMilliseconds = sw.ElapsedMilliseconds;
        return result;
    }

    public async Task<OperateResult> WriteRawAsync(
        string address, byte[] data, CancellationToken ct = default)
    {
        if (State != ConnectionState.Connected)
            return OperateResult.Fail(ErrorCode.CommNotConnected, "Device not connected.");

        var sw = Stopwatch.StartNew();
        var result = await WithRetry(() => WriteRawCoreAsync(address, data, ct),
                                     Options.RetryCount, ct);
        sw.Stop();

        Statistics.TotalWrites++;
        if (result.IsSuccess)
        {
            Statistics.SuccessWrites++;
            Statistics.TotalBytesSent += data.Length;
            Statistics.LastSuccessTime = DateTime.Now;
        }
        else
        {
            Statistics.FailedWrites++;
            OnError(result);
        }
        result.ElapsedMilliseconds = sw.ElapsedMilliseconds;
        return result;
    }

    // 默认实现：通过 ReadRaw + ByteTransform 解析
    public virtual async Task<OperateResult<TagValue>> ReadAsync(
        Tag tag, CancellationToken ct = default)
    {
        var (length, _) = GetByteLength(tag);
        var raw = await ReadRawAsync(tag.Address, length, ct);
        if (!raw.IsSuccess) return OperateResult.Fail<TagValue>(raw);

        var value = ParseValue(tag, raw.Content!);
        return OperateResult.Ok(new TagValue
        {
            TagName = tag.Name,
            Value = value,
            Quality = TagQuality.Good,
            Timestamp = DateTime.Now,
            RawBytes = raw.Content
        });
    }

    public virtual async Task<OperateResult<T>> ReadAsync<T>(
        string address, CancellationToken ct = default)
    {
        var tag = new Tag { Address = address, DataType = ResolveDataType<T>() };
        var r = await ReadAsync(tag, ct);
        if (!r.IsSuccess) return OperateResult.Fail<T>(r);
        return OperateResult.Ok((T)r.Content!.Value!);
    }

    public virtual async Task<OperateResult> WriteAsync(
        Tag tag, object value, CancellationToken ct = default)
    {
        var bytes = SerializeValue(tag, value);
        return await WriteRawAsync(tag.Address, bytes, ct);
    }

    public virtual Task<OperateResult> WriteAsync<T>(
        string address, T value, CancellationToken ct = default)
    {
        var tag = new Tag { Address = address, DataType = ResolveDataType<T>() };
        return WriteAsync(tag, value!, ct);
    }

    // 默认实现：循环单点读，子类应重写为合并地址的优化版
    public virtual async Task<OperateResult<IReadOnlyList<TagValue>>> ReadBatchAsync(
        IEnumerable<Tag> tags, CancellationToken ct = default)
    {
        var list = new List<TagValue>();
        foreach (var tag in tags)
        {
            ct.ThrowIfCancellationRequested();
            var r = await ReadAsync(tag, ct);
            list.Add(r.IsSuccess
                ? r.Content!
                : new TagValue { TagName = tag.Name, Quality = TagQuality.Bad });
        }
        return OperateResult.Ok<IReadOnlyList<TagValue>>(list);
    }

    public virtual async Task<OperateResult> WriteBatchAsync(
        IEnumerable<KeyValuePair<Tag, object>> writes, CancellationToken ct = default)
    {
        foreach (var kv in writes)
        {
            ct.ThrowIfCancellationRequested();
            var r = await WriteAsync(kv.Key, kv.Value, ct);
            if (!r.IsSuccess) return r;
        }
        return OperateResult.Ok();
    }

    // ----------- 工具方法 -----------
    protected void ChangeState(ConnectionState newState)
    {
        if (State == newState) return;
        var old = State;
        State = newState;
        Logger.Info($"Device [{DeviceId}] state: {old} → {newState}");
        StateChanged?.Invoke(this, new ConnectionStateChangedEventArgs(old, newState));
    }

    private void OnError(OperateResult result) =>
        ErrorOccurred?.Invoke(this,
            new CommunicationErrorEventArgs(result.ErrorCode, result.Message, result.Exception));

    private async Task<OperateResult<T>> WithRetry<T>(
        Func<Task<OperateResult<T>>> action, int retry, CancellationToken ct)
    {
        OperateResult<T> last = null!;
        for (var i = 0; i <= retry; i++)
        {
            ct.ThrowIfCancellationRequested();
            last = await action();
            if (last.IsSuccess) return last;
            if (i < retry) await Task.Delay(Options.RetryIntervalMs, ct);
        }
        return last;
    }

    private async Task<OperateResult> WithRetry(
        Func<Task<OperateResult>> action, int retry, CancellationToken ct)
    {
        OperateResult last = null!;
        for (var i = 0; i <= retry; i++)
        {
            ct.ThrowIfCancellationRequested();
            last = await action();
            if (last.IsSuccess) return last;
            if (i < retry) await Task.Delay(Options.RetryIntervalMs, ct);
        }
        return last;
    }

    private void StartHeartbeat()
    {
        StopHeartbeat();
        if (Options.HeartbeatIntervalMs <= 0) return;
        _heartbeatCts = new CancellationTokenSource();
        _ = Task.Run(async () =>
        {
            var token = _heartbeatCts.Token;
            while (!token.IsCancellationRequested)
            {
                try
                {
                    await Task.Delay(Options.HeartbeatIntervalMs, token);
                    var r = await PingAsync(token);
                    if (!r.IsSuccess && State == ConnectionState.Connected)
                    {
                        Logger.Warn($"[{DeviceId}] heartbeat failed, reconnecting...");
                        ChangeState(ConnectionState.Reconnecting);
                        await ConnectAsync(token);
                    }
                }
                catch (OperationCanceledException) { break; }
                catch (Exception ex) { Logger.Warn($"[{DeviceId}] heartbeat error", ex); }
            }
        });
    }

    private void StopHeartbeat()
    {
        _heartbeatCts?.Cancel();
        _heartbeatCts?.Dispose();
        _heartbeatCts = null;
    }

    private void UpdateAvgLatency(long ms)
    {
        Statistics.AverageReadLatencyMs =
            (Statistics.AverageReadLatencyMs * (Statistics.SuccessReads - 1) + ms)
            / Math.Max(1, Statistics.SuccessReads);
    }

    protected abstract (ushort byteLength, ushort registerCount) GetByteLength(Tag tag);
    protected abstract object ParseValue(Tag tag, byte[] bytes);
    protected abstract byte[] SerializeValue(Tag tag, object value);
    protected abstract DataType ResolveDataType<T>();

    public async ValueTask DisposeAsync()
    {
        await DisconnectAsync();
        _connLock.Dispose();
        GC.SuppressFinalize(this);
    }
}
```

## 六、点位订阅服务

```csharp
namespace Acme.Industrial.Communication.Core;

public class TagSubscriptionService : ITagSubscriber
{
    private readonly IDeviceDriver _driver;
    private readonly IAppLogger _logger;
    private readonly ConcurrentDictionary<Guid, SubscriptionEntry> _subs = new();
    private readonly Timer _scanTimer;
    private readonly ConcurrentDictionary<string, object?> _lastValues = new();

    public TagSubscriptionService(IDeviceDriver driver, IAppLoggerFactory loggerFactory)
    {
        _driver = driver;
        _logger = loggerFactory.CreateLogger<TagSubscriptionService>();
        _scanTimer = new Timer(OnScan, null, Timeout.Infinite, Timeout.Infinite);
    }

    public IDisposable Subscribe(IEnumerable<Tag> tags, Action<TagValue> onValueChanged)
    {
        var id = Guid.NewGuid();
        _subs[id] = new SubscriptionEntry(tags.ToList(), onValueChanged);
        EnsureScanning();
        return new ActionDisposable(() => _subs.TryRemove(id, out _));
    }

    public void UnsubscribeAll()
    {
        _subs.Clear();
        _scanTimer.Change(Timeout.Infinite, Timeout.Infinite);
    }

    private void EnsureScanning() => _scanTimer.Change(0, 100); // 100ms 调度精度

    private async void OnScan(object? _)
    {
        try
        {
            var due = _subs.Values
                .SelectMany(s => s.Tags)
                .GroupBy(t => t.ScanRate)
                .ToList();

            foreach (var group in due)
            {
                var tags = group.ToList();
                var result = await _driver.ReadBatchAsync(tags);
                if (!result.IsSuccess) continue;

                foreach (var tv in result.Content!)
                {
                    if (!HasChanged(tv, FindTag(tags, tv.TagName))) continue;
                    _lastValues[tv.TagName] = tv.Value;
                    foreach (var sub in _subs.Values)
                        sub.Callback(tv);
                }
            }
        }
        catch (Exception ex) { _logger.Warn("Scan error", ex); }
    }

    private bool HasChanged(TagValue newValue, Tag tag)
    {
        if (!_lastValues.TryGetValue(newValue.TagName, out var old)) return true;
        if (tag.DeadBand <= 0) return !Equals(old, newValue.Value);
        try
        {
            var oldD = Convert.ToDouble(old);
            var newD = Convert.ToDouble(newValue.Value);
            return Math.Abs(oldD - newD) > tag.DeadBand;
        }
        catch { return !Equals(old, newValue.Value); }
    }

    private static Tag FindTag(IEnumerable<Tag> tags, string name) =>
        tags.First(t => t.Name == name);

    private record SubscriptionEntry(List<Tag> Tags, Action<TagValue> Callback);

    private class ActionDisposable : IDisposable
    {
        private readonly Action _action;
        public ActionDisposable(Action a) => _action = a;
        public void Dispose() => _action();
    }

    public void Dispose() => _scanTimer.Dispose();
}
```

```csharp
/// HMI 运行时引擎，协调数据订阅、动画播放和脚本执行。

public class HmiRuntimeEngine
{
    private readonly IDeviceManager _deviceManager;
    private readonly IScriptManager _scriptManager;
    private readonly HmiAnimationEngine _animationEngine;
    private readonly TagSubscriptionManager _subscriptionManager;

    public HmiRuntimeEngine(HmiRuntimeServices services)
    {
        _deviceManager = services.DeviceManager;
        _scriptManager = services.ScriptManager;
        _animationEngine = new HmiAnimationEngine();
        _subscriptionManager = new TagSubscriptionManager(_deviceManager);
    }

    /// <summary>
    /// 订阅页面中所有元素的点位绑定。
    /// </summary>
    public void SubscribePage(HmiPage page, Action<HmiElement, TagValue> onValueChanged)
    {
        foreach (var element in page.Elements)
        {
            foreach (var binding in element.TagBindings)
            {
                if (!binding.IsEnabled) continue;
                
                var device = _deviceManager.GetDevice(binding.DeviceId);
                if (device == null) continue;

                var tag = new Tag
                {
                    Name = binding.FullTagName,
                    Address = binding.TagAddress,
                    ScanRate = binding.ScanRate > 0 ? binding.ScanRate : 1000
                };

                _subscriptionManager.Subscribe(tag, tv =>
                {
                    onValueChanged(element, tv);
                    ApplyBindingAndAnimation(element, binding, tv);
                });
            }
        }
    }

    /// <summary>
    /// 应用绑定值并触发相关动画。
    /// </summary>
    private void ApplyBindingAndAnimation(HmiElement element, TagBinding binding, TagValue value)
    {
        // 应用转换器
        object? displayValue = value.Value;
        if (!string.IsNullOrEmpty(binding.Converter))
        {
            displayValue = ApplyConverter(binding.Converter, value.Value, binding.ConverterParameter);
        }

        // 更新元素属性
        switch (binding.TargetProperty)
        {
            case "Value":
                if (element is DigitalDisplayElement display)
                    display.Text = FormatValue(displayValue, display.Format);
                break;
                
            case "FillColor":
            case "Color":
                if (displayValue is bool b)
                    element.FillColor = b ? "#00FF00" : "#333333";
                else if (displayValue is string colorStr)
                    element.FillColor = colorStr;
                break;
                
            case "Visibility":
                element.IsVisible = Convert.ToBoolean(displayValue);
                break;
        }

        // 触发相关动画
        foreach (var animation in element.Animations)
        {
            if (!animation.IsEnabled) continue;
            
            var shouldPlay = EvaluateAnimationTrigger(animation, value);
            if (shouldPlay)
            {
                _animationEngine.PlayAnimation(element, animation);
            }
        }
    }

    /// <summary>
    /// 评估动画触发条件。
    /// </summary>
    private bool EvaluateAnimationTrigger(HmiAnimation animation, TagValue value)
    {
        return animation.Trigger switch
        {
            AnimationTrigger.ValueChanged => true,
            AnimationTrigger.BoolTrue => value.Value is bool b && b,
            AnimationTrigger.BoolFalse => value.Value is bool b2 && !b2,
            AnimationTrigger.QualityGood => value.Quality == TagQuality.Good,
            AnimationTrigger.QualityBad => value.Quality != TagQuality.Good,
            _ => false
        };
    }

    private object? ApplyConverter(string converterName, object? value, object? parameter)
    {
        return converterName switch
        {
            "BoolToColor" => (bool)value! ? "#00FF00" : "#333333",
            "InverseBool" => !(bool)value!,
            "RangeConverter" => Convert.ToDouble(value) / 100.0,
            "NumberFormat" => FormatValue(value, parameter?.ToString()),
            _ => value
        };
    }

    private string FormatValue(object? value, string? format)
    {
        if (value == null) return "****";
        if (!string.IsNullOrEmpty(format))
            return string.Format($"{{0:{format}}}", value);
        return value.ToString() ?? "";
    }

    /// <summary>
    /// 执行元素点击脚本。
    /// </summary>
    public async Task ExecuteElementScriptAsync(string? script, HmiElement element)
    {
        if (string.IsNullOrEmpty(script)) return;
        
        // 创建元素上下文
        var elementContext = new ElementScriptContext(this, element);
        await _scriptManager.ExecuteScriptAsync(script, elementContext);
    }

    /// <summary>
    /// 写入点位（供脚本和操作调用）。
    /// </summary>
    public async Task<OperateResult> WriteTagAsync(string deviceId, string address, object value)
    {
        var device = _deviceManager.GetDevice(deviceId);
        if (device == null)
            return OperateResult.Fail(ErrorCode.CommNotConnected, $"设备不存在: {deviceId}");
        
        return await device.WriteAsync(address, value);
    }

    /// <summary>
    /// 读取点位。
    /// </summary>
    public async Task<TagValue?> ReadTagAsync(string deviceId, string address)
    {
        var device = _deviceManager.GetDevice(deviceId);
        if (device == null) return null;
        
        var result = await device.ReadAsync(address);
        return result.IsSuccess ? result.Content : null;
    }
}
```

```csharp
        _devices = devices;
        _bus = bus;
    }

    public async Task StartAsync()
    {
        await _devices.AddDeviceAsync(new ConnectionOptions
        {
            DeviceId = "Reactor-01",
            DeviceName = "1号反应釜",
            Protocol = "ModbusTcp",
            Host = "192.168.1.10",
            Port = 502,
            SlaveAddress = 1,
            HeartbeatIntervalMs = 3000
        });

        var device = _devices.GetDevice("Reactor-01")!;
        await device.ConnectAsync();

        var temp = await device.ReadAsync<float>("40001");
        if (temp.IsSuccess && temp.Content > 80)
            await _bus.PublishAsync(new AlarmTriggeredEvent
            {
                AlarmId = "Reactor01.HighTemp",
                Level = AlarmLevel.High,
                Source = "Reactor-01",
                Message = $"温度过高: {temp.Content}℃"
            });
    }
}
```

## 十二、关键设计决策说明

| 决策 | 原因 |
|------|------|
| **统一 IDeviceDriver** | 业务层只学一套 API，更换协议无需改业务代码 |
| **基类封装连接/重试/统计/心跳** | 各协议实现只需关注协议本身的 PDU 编解码 |
| **OperateResult 不抛异常** | 工业现场异常太频繁（断线、超时），异常成本高且易漏处理 |
| **IByteTransform 抽象字节序** | 字节序错乱是工业通讯最常见 bug，集中处理避免重复 |
| **批量读 + 地址合并** | 单点读 1000 次 vs 合并后 10 次 = 100 倍性能差距 |
| **订阅式刷新 + 死区过滤** | 减少无意义事件，配合 EventBus 解耦 UI 与采集 |
| **Mock 驱动** | 没设备也能开发、CI 也能跑测试，强烈推荐 |
| **设备 ID + 协议名** | 一台机器同时连多种 PLC 是常态，必须统一管理 |
| **脚本引擎集成** | 通过 `IndustrialScriptGlobals` 注入点位访问能力 |
| **HMI 数据绑定** | 通过 `TagSubscriptionManager` 实现实时数据更新 |

## 十三、扩展方向

落地这一层后，还可继续扩展：

- **OPC UA Server**：把采集的数据再以 OPC UA 暴露给 MES/SCADA
- **MQTT Bridge**：自动把点位变化推送到云平台
- **历史库（Historian）**：基于订阅事件自动写入时序库
- **冗余/双机热备**：主备设备自动切换
- **设备模板**：复用同型号设备的点位定义
- **协议解析器可视化工具**：抓包 → 解析 → 调试
- **HMI 图元库扩展**：支持更多工业控件和自定义图元
- **脚本 IDE**：集成代码编辑、调试、版本管理

## 十四、小结

通讯层是工业框架的"地基"。把这层抽象做扎实后：

- 上层业务（脚本引擎、HMI 系统）专注于"做什么"，不关心"怎么通讯"
- 新增协议只是新增一个类，不影响其他模块
- 通过 Mock 驱动实现 100% 可测试
- 统计、日志、重连、心跳一次搞定，所有协议自动具备
- 脚本引擎和 HMI 系统通过统一接口访问点位，实现无缝集成

至此，目录结构 → 核心接口 → 通讯层抽象 → 脚本引擎 → HMI 系统 五大基础已经齐备，可以开始动手实现框架的第一个版本了。
